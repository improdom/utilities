private List<BenchmarkSettings> SplitQueriesIntoChunks(BenchmarkSettings originalSettings, int chunkSize)
{
    var splitSettingsList = new List<BenchmarkSettings>();
    
    // Split Queries into chunks
    var queryChunks = originalSettings.Queries
        .Select((query, index) => new { query, index })
        .GroupBy(x => x.index / chunkSize)
        .Select(g => g.Select(x => x.query).ToList())
        .ToList();

    // Create new BenchmarkSettings for each chunk
    foreach (var chunk in queryChunks)
    {
        var newSettings = new BenchmarkSettings
        {
            Benchmark = originalSettings.Benchmark,
            Queries = chunk,
            Parameters = originalSettings.Parameters
        };
        splitSettingsList.Add(newSettings);
    }

    return splitSettingsList;
}
