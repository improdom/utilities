# List of items
$list = "A,B,C,D,E,F"

# Convert the list to an array
$items = $list -split ','

# Loop through each item
foreach ($item in $items) {
    # Create the payload with 'name' and 'processName'
    $body = @{
        "name" = $item
        "processName" = "update"
    } | ConvertTo-Json

    # Define the headers (if needed)
    $headers = @{
        "Content-Type" = "application/json"
    }

    # Send the POST request
    $response = Invoke-RestMethod -Uri "https://localhost/process" -Method Post -Body $body -Headers $headers

    # Output the response
    Write-Host "Response for item $item: $response"
}
