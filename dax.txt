using System;
using System.Text.Json.Serialization;

public class ReadinessEventMessage

    [JsonPropertyName("sourceSystem")]

    [JsonPropertyName("subDesk")]

    [JsonPropertyName("datasetName")]
    public string DatasetName { get; set; }

    [JsonPropertyName("workspaceName")]
    public string WorkspaceName { get; set; }

    [JsonPropertyName("tenantId")]
    public string TenantId { get; set; }

    [JsonPropertyName("clientId")]
    public string ClientId { get; set; }

    [JsonPropertyName("secretStore")]
    public string SecretStore { get; set; }

    [JsonPropertyName("cobDate")]

    [JsonPropertyName("isRetry")]
    public bool IsRetry { get; set; }

    [JsonPropertyName("dbxInstance")]
    public string DbxInstance { get; set; }

    [JsonPropertyName("dbxPath")]
    public string DbxPath { get; set; }

    [JsonPropertyName("dbxDatabase")]
    public string DbxDatabase { get; set; }

    [JsonPropertyName("modelName")]
    public string ModelName { get; set; }

    public string ReadinessCategory { get; set; }

    [JsonPropertyName("readinessSubCategory")]
    public string ReadinessSubCategory { get; set; }

    [JsonPropertyName("readinessComponent")]
    public string ReadinessComponent { get; set; }

    [JsonPropertyName("readinessStatus")]
    public string ReadinessStatus { get; set; }
}
