TV Change EQ Hedge Funds Slide +/-30% =
VAR StressMag = SELECTEDVALUE( 'Scenario'[Stress Magnitude] )
VAR MinValue =
    MINX(
        FILTER(
            CROSSJOIN(
                ALL( 'Scenario'[Scenario Name] ), 
                ALL( 'Scenario'[Stress Magnitude] )
            ),
            NOT( 'Scenario'[Stress Magnitude] IN { -30, 30 } )
        ),
        [TV Change EQ Hedge Funds]
    )
RETURN
    IF(
        ISBLANK( MinValue ), 
        BLANK(), -- Equivalent to NULL in MDX
        IF(
            StressMag IN { -30, 30 }, 
            BLANK(), -- Scope logic to remove -30% and 30%
            MinValue
        )
    )
