using System.Text.RegularExpressions;

public class DaxColumnInfo
{
    public string TableName { get; set; }
    public string ColumnName { get; set; }
}

public static DaxColumnInfo ExtractTableAndColumn(string input)
{
    var match = Regex.Match(input, @"^(.*?)\[(.*)\]$");

    if (match.Success && match.Groups.Count == 3)
    {
        return new DaxColumnInfo
        {
            TableName = match.Groups[1].Value.Trim(),
            ColumnName = match.Groups[2].Value.Trim()
        };
    }

    return null; // or handle invalid format appropriately
}
